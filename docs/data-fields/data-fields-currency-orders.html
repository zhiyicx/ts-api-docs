# 货币订单 currency_orders

字段 类型 说明
id int 自增id
owner_id int 订单所属者
title varchar 订单标题
body text 订单内容
type tinyint 1：入账、-1：支出
target_type varchar 操作类型
target_id int 目标id
currency int 货币别名id, 关联currency_types主键
amount int 订单金额
state tinyint 订单状态，0: 等待，1：成功，-1: 失败
created_at timestamp 创建时间
updated_at timestamp 最后更新时间
